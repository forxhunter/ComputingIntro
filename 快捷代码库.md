## 牛顿迭代法

```python
# Newton's method
def solve(function = "x**3-5*x**2+10*x-80", a=4.0):
    def f(x):
        return eval(function)

    def df(x, dx):
        return (f(x + dx) - f(x)) / dx

    while abs(f(a)) > 1e-10:
        if df(a, 1e-10) == 0:
            a -= 1e-10
        else:
            a = a - f(a) / df(a, 1e-10)
    return a


ans = solve()
print("{:.9f}".format(ans))
```



## 自定义迭代器

```python
class Squares:
    def __init__(self, start, stop):  # 迭代起始、终止位
        self.value = start
        self.stop = stop

    def __iter__(self):     # 返回自身的迭代器
        return self

    def __next__(self):     # 返回下一个元素
        if self.value > self.stop:   # 结尾时抛出异常
            raise (StopIteration)
        item = self.value**2
        self.value += 1
        return item

if __name__ == "__main__":
    for i in Squares(1, 5):
        print(i, end=" ")

    s = Squares(1,5)
    print()
    print(9 in s)
```

```python
    queue = []  # 初始化一个空队列
    queue.append(s) # 将所有节点入队列
    seen = set()
    seen.add(s)
    parent = {s : None}

    while(len(queue) > 0):
        vertex = queue.pop(0)
        nodes = graph[vertex]
        for w in nodes:
            if w not in seen:
                queue.append(w)
                seen.add(w)

    return parent
```

(注意邻接是4点邻接还是8点邻接)

以四格遍历为例，输出连成片的地区：r地区、b地区、#空

```python
black = 'b'
red = 'r'
empty = '#'
neighbour = [[0, -1], [0, 1], [1, 0], [-1, 0]]


def dfs(start, area, visited, color):
    r, c = start
    visited[r][c] = 1
    stack = []
    stack.append([r, c, -1])
    while len(stack) != 0:
        x, y, previous_n = stack[-1]
        if visited[x][y] != 1:
            visited[x][y] = 1
        # judge if remove in the end
        remove_from_stack = True

        for dire in range(4):
            if dire != previous_n:
                dx, dy = neighbour[dire]
                temp_x = x + dx
                temp_y = y + dy
                # have children node, not remove now
                if area[temp_x][temp_y] == color and visited[temp_x][temp_y] == 0:
                    remove_from_stack = False
                    parent = neighbour.index([-dx, -dy])
                    stack.append([temp_x, temp_y, parent])

        # decide if remove
        if remove_from_stack is True:
            stack.pop()

    return


# count areas

c_red = 0

n = int(input())
# 初始化

visited = []
```

### 有向图

#### 有向图bfs

+ 字典列链表形式

  求最大连通子图的节点数为例

  ```python
  ## create graph 字典类型，有向图
  graph = {}
  ids = set()
  
  for _ in range(int(input())):
      ln = input().split(':')
      u = int(ln[0].rstrip())
      ids.add(u)
      if u not in graph:
          neighbours = [int(_) for _ in ln[1].split()]
          graph[u] = neighbours
  # bfs: graph(dict type),initial node code
  def bfs(graph, initial):    
      visited = []    
      queue = [initial]
   
      while queue:        
          node = queue.pop(0)
          if node not in visited:            
              visited.append(node)
              
              if node not in graph:
                  continue
              
              for nei in graph[node]:       # neighbour
                  queue.append(nei)
      return visited
  
  ## formal code: bfs travel
  maxp = 0
  for i in ids:
      bfs_path = bfs(graph, i)
      if -1 in bfs_path:
          bfs_path.remove(-1)
      maxp = max(maxp, len(bfs_path))
      #print(len(bfs_path), bfs_path)
  print(maxp)
  ```

  **input sample**

  **非递归实现**

  ```python
  # 读取数据： 字典类型，有向图
  graph = {}
  ids = set()
  
  for _ in range(int(input())):
      ln = input().split(':')
      u = int(ln[0].rstrip())
      ids.add(u)
      if u not in graph:
          neighbours = [int(_) for _ in ln[1].split()]
          graph[u] = neighbours
  
  def dfs_Non_recursion(graph, initial):
      visited = [initial]
      stack = [initial]
      while stack:
          node = stack[-1]
          if node not in visited:
              visited.append(node)
          
          if node not in graph:
              stack.pop()
              continue
          
          remove_from_stack = True       
          for nei in graph[node]:         # neighbour
              if nei not in visited:
                  stack.append(nei)
                  remove_from_stack = False
                  break
          if remove_from_stack:
              stack.pop()
      return visited
  
  
  ## dfs_Non_recursion travel
  maxp = 0
  for i in ids:
      dfs_path = dfs_Non_recursion(graph, i)
      if -1 in dfs_path:
          dfs_path.remove(-1)
      maxp = max(maxp, len(dfs_path))
      #print(len(bfs_path), bfs_path) #返回每次bfs路径长度，和具体过程
  print(maxp)	#返回最大路径长度
  ```

  ```python
  # 读取数据： 字典类型，有向图
  graph = {}
  ```
